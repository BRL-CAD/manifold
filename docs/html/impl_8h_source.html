68 template <typename Precision, typename I>
69 Impl(const MeshGLP<Precision, I>& meshGL) {
70 const uint32_t numVert = meshGL.NumVert();
71 const uint32_t numTri = meshGL.NumTri();
72
73 if (meshGL.numProp < 3) {
74 MarkFailure(Error::MissingPositionProperties);
75 return;
76 }
77
78 if (meshGL.mergeFromVert.size() != meshGL.mergeToVert.size()) {
79 MarkFailure(Error::MergeVectorsDifferentLengths);
80 return;
81 }
82
83 if (!meshGL.runTransform.empty() &&
84 12 * meshGL.runOriginalID.size() != meshGL.runTransform.size()) {
85 MarkFailure(Error::TransformWrongLength);
86 return;
87 }
88
89 if (!meshGL.runOriginalID.empty() && !meshGL.runIndex.empty() &&
90 meshGL.runOriginalID.size() + 1 != meshGL.runIndex.size() &&
91 meshGL.runOriginalID.size() != meshGL.runIndex.size()) {
92 MarkFailure(Error::RunIndexWrongLength);
93 return;
94 }
95
96 if (!meshGL.faceID.empty() && meshGL.faceID.size() != meshGL.NumTri()) {
97 MarkFailure(Error::FaceIDWrongLength);
98 return;
99 }
100
101 std::vector<int> prop2vert(numVert);
102 std::iota(prop2vert.begin(), prop2vert.end(), 0);
103 for (size_t i = 0; i < meshGL.mergeFromVert.size(); ++i) {
104 const uint32_t from = meshGL.mergeFromVert[i];
105 const uint32_t to = meshGL.mergeToVert[i];
106 if (from >= numVert || to >= numVert) {
107 MarkFailure(Error::MergeIndexOutOfBounds);
108 return;
109 }
110 prop2vert[from] = to;
111 }
112
113 const auto numProp = meshGL.numProp - 3;
114 meshRelation_.numProp = numProp;
115 meshRelation_.properties.resize(meshGL.NumVert() * numProp);
116 // This will have unreferenced duplicate positions that will be removed by
117 // Impl::RemoveUnreferencedVerts().
118 vertPos_.resize(meshGL.NumVert());
119
120 for (size_t i = 0; i < meshGL.NumVert(); ++i) {
121 for (const int j : {0, 1, 2})
122 vertPos_[i][j] = meshGL.vertProperties[meshGL.numProp * i + j];
123 for (size_t j = 0; j < numProp; ++j)
124 meshRelation_.properties[i * numProp + j] =
125 meshGL.vertProperties[meshGL.numProp * i + 3 + j];
126 }
127
128 halfedgeTangent_.resize(meshGL.halfedgeTangent.size() / 4);
129 for (size_t i = 0; i < halfedgeTangent_.size(); ++i) {
130 for (const int j : {0, 1, 2, 3})
131 halfedgeTangent_[i][j] = meshGL.halfedgeTangent[4 * i + j];
132 }
133
134 Vec<TriRef> triRef;
135 if (!meshGL.runOriginalID.empty()) {
136 auto runIndex = meshGL.runIndex;
137 const auto runEnd = meshGL.triVerts.size();
138 if (runIndex.empty()) {
139 runIndex = {0, static_cast<I>(runEnd)};
140 } else if (runIndex.size() == meshGL.runOriginalID.size()) {
141 runIndex.push_back(runEnd);
142 }
143 triRef.resize(meshGL.NumTri());
144 const auto startID = Impl::ReserveIDs(meshGL.runOriginalID.size());
145 for (size_t i = 0; i < meshGL.runOriginalID.size(); ++i) {
146 const int meshID = startID + i;
147 const int originalID = meshGL.runOriginalID[i];
148 for (size_t tri = runIndex[i] / 3; tri < runIndex[i + 1] / 3; ++tri) {
149 TriRef& ref = triRef[tri];
150 ref.meshID = meshID;
151 ref.originalID = originalID;
152 ref.tri = meshGL.faceID.empty() ? tri : meshGL.faceID[tri];
153 }
154
155 if (meshGL.runTransform.empty()) {
156 meshRelation_.meshIDtransform[meshID] = {originalID};
157 } else {
158 const Precision* m = meshGL.runTransform.data() + 12 * i;
159 meshRelation_.meshIDtransform[meshID] = {
160 originalID,
161 {m[0], m[1], m[2], m[3], m[4], m[5], m[6], m[7], m[8], m[9],
162 m[10], m[11]}};
163 }
164 }
165 }
166
167 Vec<ivec3> triVerts;
168 triVerts.reserve(numTri);
169 for (size_t i = 0; i < numTri; ++i) {
170 ivec3 tri;
171 for (const size_t j : {0, 1, 2}) {
172 uint32_t vert = (uint32_t)meshGL.triVerts[3 * i + j];
173 if (vert >= numVert) {
174 MarkFailure(Error::VertexOutOfBounds);
175 return;
176 }
177 tri[j] = prop2vert[vert];
178 }
179 if (tri[0] != tri[1] && tri[1] != tri[2] && tri[2] != tri[0]) {
180 triVerts.push_back(tri);
181 if (triRef.size() > 0) {
182 meshRelation_.triRef.push_back(triRef[i]);
183 }
184 if (numProp > 0) {
185 meshRelation_.triProperties.push_back(
186 ivec3(static_cast<uint32_t>(meshGL.triVerts[3 * i]),
187 static_cast<uint32_t>(meshGL.triVerts[3 * i + 1]),
188 static_cast<uint32_t>(meshGL.triVerts[3 * i + 2])));
189 }
190 }
191 }
192
193 CreateHalfedges(triVerts);
194 if (!IsManifold()) {
195 MarkFailure(Error::NotManifold);
196 return;
197 }
198
199 CalculateBBox();
200 if (!IsFinite()) {
201 MarkFailure(Error::NonFiniteVertex);
202 return;
203 }
204 SetPrecision(meshGL.precision);
205
206 SplitPinchedVerts();
207
208 CalculateNormals();
209
210 if (meshGL.runOriginalID.empty()) {
211 InitializeOriginal();
212 }
213
214 SimplifyTopology();
215 Finish();
216
217 // A Manifold created from an input mesh is never an original - the input is
218 // the original.
219 meshRelation_.originalID = -1;
220 }

226 inline void ForVert(int halfedge, std::function<void(int halfedge)> func) {
227 int current = halfedge;
228 do {
229 current = NextHalfedge(halfedge_[current].pairedHalfedge);
230 func(current);
231 } while (current != halfedge);
232 }
233
234 template <typename T>
235 void ForVert(
236 int halfedge, std::function<T(int halfedge)> transform,
237 std::function<void(int halfedge, const T& here, T& next)> binaryOp) {
238 T here = transform(halfedge);
239 int current = halfedge;
240 do {
241 const int nextHalfedge = NextHalfedge(halfedge_[current].pairedHalfedge);
242 T next = transform(nextHalfedge);
243 binaryOp(current, here, next);
244 here = next;
245 current = nextHalfedge;
246 } while (current != halfedge);
247 }
